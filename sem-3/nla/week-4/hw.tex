                    \|A\|_F 
                    &= \left(\sum_{i=1}^m\sum_{j=1}^n|a_{ij}|^2\right)^\frac{1}{2} &\implies \\
                    \|A\|_F^2
                    &= \sum_{i=1}^m\sum_{j=1}^n|a_{ij}|^2 \\
                    &= \sum_{j=1}^n\left(\sum_{i=1}^m|a_{ij}|^2\right) \\
                    &= \sum_{j=1}^n\left(\left(\sum_{i=1}^m|a_{ij}|^2\right)^\frac{1}{2}\right)^2 \\

                \end{aligned}
